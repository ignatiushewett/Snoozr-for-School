-- SQL to load this user's sessions (READ):
SELECT s.id, s.date, s.duration_mins, s.notes, sub.name
FROM sessions s
JOIN subjects sub ON s.subject_id = sub.id
WHERE s.user_id = :current_user_id
ORDER BY s.date DESC;
